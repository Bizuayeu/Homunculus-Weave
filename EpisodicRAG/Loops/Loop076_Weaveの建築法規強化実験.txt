function checkOneRoomRegulation(project) {
  // 区名とワンルームキーワードで検索
  const query = `site:jourei.jyunpo.com ${project.土地所在} ワンルーム 住戸 戸数`;
  const results = webSearch(query);
  
  // 大田区の例

    if (project.計画戸数 >= 15) {
      return {
        規制対象: true,
        理由: "15戸以上の集団住宅",
        追加設置義務: ["ファミリー住戸", "管理人室", "集会室"],
        根拠: "大田区開発指導要綱"
      };
    }
  }
  
  return {規制対象: false};
}
